52template<class T> void SubSampleTo( const vector<T> & pFrom, vector<T> & pTo, size_t pSampleSize )
53{
54 // Get sampled indices
55 vector<size_t> lIdx;
56 GetSubSampIdx( pFrom.size(), pSampleSize, lIdx );

58 // Copy sampled elements
59 pTo.resize( pSampleSize );
60 for( size_t i=0 ; i<lIdx.size() ; i++ )
